Defaults

Click the button to start a progress indicator.

Progress:

This example shows the Progress component "out-of-the-box," using only defaults. No options are specified.

  • JavaScript
  • HTML
  • CSS
var timer = support.timeSimulator(0);
$("#showButton1").click(function () {
    support.activateProgressBar(basicBar, timer);
});

var basicBar = fluid.progress("#progress-demo-1");
<div id="progress-demo-1" class="demo-container">
    <button id="showButton1">Show...</button>
    <div class="flc-progress demo-progress">Progress:
        <div class="flc-progress-bar demo-bar">
            <div class="flc-progress-indicator demo-indicator"></div>
        </div>
    </div>
</div>

.demo-container {
    height: 70px;
    width: 300px;
}
.demo-bar {
    border: 1px solid #2D3E76;
    width: 250px;
}
.demo-indicator {
    height: 16px;
    width: 5px;
    border: 1px solid #2D3E76;
    background-color: #7B7EAE;
}
